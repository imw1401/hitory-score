  <script>
    // 1. 초기 세팅
    lucide.createIcons(); // 아이콘 렌더링

    // 학년별 수행평가 항목 정의
    const assessmentNames = {
      "2": ["인물 보고서", "역사신문", "학습지 포트폴리오"],
      "3": ["문화유산 홍보하기", "드라마 기획안 만들기", "풍자화 만들기"]
    };

    // 데이터베이스 역할 (로컬 스토리지 사용)
    let records = JSON.parse(localStorage.getItem("historyScores")) || [];

    // 화면 하단 알림창(토스트) 띄우기 함수
    function showToast(message) {
      const toast = document.getElementById("toast");
      toast.textContent = message;
      toast.classList.remove("translate-y-10", "opacity-0");
      setTimeout(() => toast.classList.add("translate-y-10", "opacity-0"), 3000);
    }
    
    // 교사 패널 저장 인원 업데이트 함수
    function updateRecordCount() {
      const badge = document.getElementById("recordCountBadge");
      if(badge) badge.textContent = `저장: ${records.length}명`;
    }

    // 2. 교사용 모드 동작 로직
    document.getElementById("toggleTeacherBtn").addEventListener("click", () => {
      document.getElementById("teacherPanel").classList.toggle("hidden");
    });

    document.getElementById("loginForm").addEventListener("submit", (e) => {
      e.preventDefault();
      const pw = document.getElementById("teacherPassword").value;
      if (pw === "teacher123") {
        document.getElementById("teacherLoginView").classList.add("hidden");
        document.getElementById("teacherContentView").classList.remove("hidden");
        updateRecordCount();
        showToast("교사 모드로 접속되었습니다.");
      } else {
        showToast("비밀번호가 틀렸습니다.");
      }
    });

    document.getElementById("logoutButton").addEventListener("click", () => {
      document.getElementById("teacherLoginView").classList.remove("hidden");
      document.getElementById("teacherContentView").classList.add("hidden");
      document.getElementById("teacherPassword").value = "";
      showToast("로그아웃 되었습니다.");
    });

    // 교사 모드: 학번 입력 시 학년에 맞는 점수 입력창 동적 생성
    const scoreInputsContainer = document.getElementById("scoreInputsContainer");
    document.getElementById("studentId").addEventListener("input", (e) => {
      const id = e.target.value;
      if (id.length >= 1) {
        const grade = id.charAt(0); // 학번 첫 글자로 2학년, 3학년 구분
        if (assessmentNames[grade]) {
          const items = assessmentNames[grade];
          scoreInputsContainer.innerHTML = items.map((item, index) => `
            <div>
              <label class="mb-1 block text-xs font-bold text-[#bc5d2d]">${item}</label>
              <input id="score${index + 1}" type="number" class="warm-focus w-full rounded-lg border border-[#e6b48a] px-3 py-2 text-sm" placeholder="점수" required>
            </div>
          `).join('');
        } else {
          scoreInputsContainer.innerHTML = `<p class="text-xs text-red-500 col-span-3 font-bold mt-2">※ 2학년 또는 3학년 학번으로 시작해야 합니다.</p>`;
        }
      } else {
        scoreInputsContainer.innerHTML = `<p class="text-xs text-[#b88b74] col-span-3">학번을 입력하면 해당 학년의 평가 항목이 나타납니다.</p>`;
      }
    });

    // 교사 모드: 점수 데이터 저장
    document.getElementById("scoreForm").addEventListener("submit", (e) => {
      e.preventDefault();
      const sId = document.getElementById("studentId").value;
      const sName = document.getElementById("studentName").value;
      const memo = document.getElementById("memo").value;
      
      const s1 = document.getElementById("score1") ? Number(document.getElementById("score1").value) : 0;
      const s2 = document.getElementById("score2") ? Number(document.getElementById("score2").value) : 0;
      const s3 = document.getElementById("score3") ? Number(document.getElementById("score3").value) : 0;

      const existingIndex = records.findIndex(r => r.studentId === sId);
      const studentData = {
        studentId: sId,
        studentName: sName,
        password: "0000", // 모든 학생 초기 비밀번호
        score1: s1,
        score2: s2,
        score3: s3,
        memo: memo
      };

      if (existingIndex !== -1) {
        records[existingIndex] = studentData; // 덮어쓰기
      } else {
        records.push(studentData); // 새 데이터 추가
      }

      localStorage.setItem("historyScores", JSON.stringify(records));
      updateRecordCount();
      showToast(`${sName} 학생의 점수가 안전하게 저장되었습니다.`);
      e.target.reset();
      scoreInputsContainer.innerHTML = `<p class="text-xs text-[#b88b74] col-span-3">학번을 입력하면 해당 학년의 평가 항목이 나타납니다.</p>`;
    });

    // 3. 학생 점수 조회 로직
    const lookupForm = document.getElementById("lookupForm");
    const lookupView = document.getElementById("lookupView");
    const resultPanel = document.getElementById("resultPanel");

    lookupForm.addEventListener("submit", (e) => {
      e.preventDefault();
      const sId = document.getElementById("lookupStudentId").value;
      const sName = document.getElementById("lookupStudentName").value;
      const sPw = document.getElementById("lookupPassword").value;

      // 기록에서 학생 찾기
      const student = records.find(r => r.studentId === sId && r.studentName === sName);

      if (!student) {
        showToast("등록된 학생 정보가 없습니다. 학번과 이름을 확인해주세요.");
        return;
      }
      if (student.password !== sPw) {
        showToast("비밀번호가 일치하지 않습니다. (초기 비밀번호: 0000)");
        return;
      }

      const grade = sId.charAt(0);
      const items = assessmentNames[grade];

      if (!items) {
        showToast("데이터 오류: 올바른 학년 정보가 아닙니다.");
        return;
      }

      // 조회 결과 화면 구성
      document.getElementById("resultName").textContent = `${student.studentName} 학생`;
      document.getElementById("resultClass").textContent = `${sId.substring(0,1)}학년 ${sId.substring(1,3)}반 ${sId.substring(3)}번`;
      document.getElementById("totalScore").textContent = student.score1 + student.score2 + student.score3;
      
      const cardsContainer = document.getElementById("scoreCardsContainer");
      const scores = [student.score1, student.score2, student.score3];
      
      cardsContainer.innerHTML = items.map((item, index) => `
        <div class="rounded-xl border border-[#e6b48a] bg-[#fffaf2] p-4 text-center shadow-sm">
          <p class="text-sm font-bold text-[#7d4b34] mb-2">${item}</p>
          <p class="text-2xl font-black text-[#bc5d2d]">${scores[index]}<span class="text-sm text-[#7d4b34] font-normal ml-1">점</span></p>
        </div>
      `).join('');

      document.getElementById("resultMemo").textContent = student.memo || "선택된 메모가 없습니다. 훌륭합니다!";

      // 화면 전환
      lookupView.classList.add("hidden");
      resultPanel.classList.remove("hidden");
    });

    document.getElementById("backBtn").addEventListener("click", () => {
      resultPanel.classList.add("hidden");
      lookupView.classList.remove("hidden");
      lookupForm.reset();
    });

  </script>
